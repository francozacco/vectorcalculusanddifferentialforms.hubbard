\section*{2 - Differentiation}
\subsection*{Basic Definitions}
\begin{proof}{\textbf{2-2.}}
Let $f:\R^2 \to \R$ be independent of the second variable i.e. for each
$x \in \R$ we have that $f(x,y_1) = f(x, y_2)$ for all $y_1, y_2 \in \R$.
\\
Then we can take some $y_0 \in \R$ and define a function $g:\R \to \R$ such
that $g(x) = f(x,y_0)$, then, for any $y\in \R$ we get that
$f(x,y) = f(x,y_0) = g(x)$ because $f$ is independent of the second variable.
\\\\
On the other hand, suppose we have a function $g:\R \to \R$ such that
$g(x) = f(x,y)$ then since this is true for any $y\in \R$ then we can write 
that
$$f(x,y_1) = g(x) = f(x, y_2)$$
for any $y_1, y_2 \in \R$, which implies that $f$ is independent of the second
variable.
\\\\
The Jacobian of $f$ at $(a,b)$ i.e. $f'(a,b)$ is given by
\begin{align*}
    f'(a,b) = (D_1f(a, b),~D_2f(a, b))
\end{align*}
Since $g(x) = f(x,y)$ then $D_1f(a,b) = D_1g(a) = g'(a)$ therefore
\begin{align*}
    f'(a,b) = (g'(a),~0)
\end{align*}
\end{proof}

\begin{proof}{\textbf{2-3.}}
A function $f:\R^2 \to \R$ is independent of the first variable if for each
$y\in \R$ we have that$f(x_1, y) = f(x_2, y)$ for all $x_1, x_2 \in \R$.
\\\\
The Jacobian $f'(a,b)$ in this case is
\begin{align*}
    f'(a,b) = (D_1f(a,b),~D_2f(a,b)) = (0,~D_2f(a,b))
\end{align*}
In the same way as we did in problem 2-2, we can show that $f$ is independent
of the first variable if and only if there is a function $g:\R \to \R$
such that $g(y) = f(x,y)$.
\\
So we get that $D_2f(a,b) = D_2g(b) = g'(b)$ and therefore
\begin{align*}
    f'(a,b) = (0, g'(b))
\end{align*}
Finally, the functions that are independent of the first and the second variable
are the functions of the form $f(x, y) = c$ where $c$ is a constant.
\end{proof}

\begin{proof}{\textbf{2-4.}}
Let us define $f:\R^2\to \R$ by
\begin{align*}
    f(x) = \begin{cases}
        |x|\cdot g\big(\frac{x}{|x|}\big) & x\neq 0\\
        0  & x = 0
    \end{cases}
\end{align*}
Where $g$ is a continuous real-valued function on the unit circle\\
$\{x \in \R^2~:~|x| = 1\}$ such that $g(0,1) = g(1,0) = 0$ and $g(-x) = -g(x)$.
\begin{itemize}
\item [(a)] If $x\in\R^2$ and $h:\R \to \R$ defined by $h(t) = f(tx)$, we want
to show that $h$ is differentiable.
\\
Suppose $x \neq 0$, also, let $a \in \R$ and let
$\lambda(k) = k|x|g(\frac{x}{|x|})$ then
\begin{align*}
    &\lim_{k \to 0} \frac{|h(a + k) - h(a) - \lambda(k)|}{|k|}
    = \lim_{k \to 0} \frac{|f((a + k)x) - f(ax) - \lambda(k)|}{|k|} =\\
    &\qquad= \lim_{k \to 0}
    \frac{1}{|k|}
    \bigg||(a + k)x|g\bigg(\frac{(a + k)x}{|(a + k)x|}\bigg)
    - |ax|g\bigg(\frac{ax}{|ax|}\bigg)
    - k|x|g\bigg(\frac{x}{|x|}\bigg)\bigg|
\end{align*}
Suppose $a + k > 0$, and suppose $a > 0$ then
\begin{align*}
    &\frac{1}{|k|}
    \bigg||(a + k)x|g\bigg(\frac{(a + k)x}{|(a + k)x|}\bigg)
    - |ax|g\bigg(\frac{ax}{|ax|}\bigg)
    - k|x|g\bigg(\frac{x}{|x|}\bigg)
    \bigg| = \\
    &= \frac{1}{|k|}
    \bigg|(a + k)|x|g\bigg(\frac{x}{|x|}\bigg)
    - a|x|g\bigg(\frac{x}{|x|}\bigg)
    - k|x|g\bigg(\frac{x}{|x|}\bigg)
    \bigg|\\
    &= 0
\end{align*}
If $a < 0$ then
\begin{align*}
    |ax|g\bigg(\frac{ax}{|ax|}\bigg)
    = -a|x|g\bigg(-\frac{ax}{a|x|}\bigg)
    = a|x|g\bigg(\frac{x}{|x|}\bigg)
\end{align*}
Where we used that $g(-x) = -g(x)$ and hence
\begin{align*}
    &\frac{1}{|k|}
    \bigg||(a + k)x|g\bigg(\frac{(a + k)x}{|(a + k)x|}\bigg)
    - |ax|g\bigg(\frac{ax}{|ax|}\bigg)
    - k|x|g\bigg(\frac{x}{|x|}\bigg)
    \bigg| =\\
    &= \frac{1}{|k|}
    \bigg|(a + k)|x|g\bigg(\frac{x}{|x|}\bigg)
    - a|x|g\bigg(\frac{x}{|x|}\bigg)
    - k|x|g\bigg(\frac{x}{|x|}\bigg)
    \bigg|\\
    &= 0
\end{align*}
Now suppose $a + k < 0$ then
\begin{align*}
    &\frac{1}{|k|}
    \bigg||(a + k)x|g\bigg(\frac{(a + k)x}{|(a + k)x|}\bigg)
    - |ax|g\bigg(\frac{ax}{|ax|}\bigg)
    - k|x|g\bigg(\frac{x}{|x|}\bigg)
    \bigg| = \\
    &= \bigg|-(a + k)|x|g\bigg(-\frac{(a + k)x}{(a + k)|x|}\bigg)
    - a|x|g\bigg(\frac{x}{|x|}\bigg)
    - k|x|g\bigg(\frac{x}{|x|}\bigg)
    \bigg|\\
    &= \bigg|(a + k)|x|g\bigg(\frac{x}{|x|}\bigg)
    - a|x|g\bigg(\frac{x}{|x|}\bigg)
    - k|x|g\bigg(\frac{x}{|x|}\bigg)
    \bigg|\\
    &= 0
\end{align*}
Finally if $a + k = 0$
\begin{align*}
    &\frac{1}{|k|}
    \bigg||(a + k)x|g\bigg(\frac{(a + k)x}{|(a + k)x|}\bigg)
    - |ax|g\bigg(\frac{ax}{|ax|}\bigg)
    - k|x|g\bigg(\frac{x}{|x|}\bigg)
    \bigg| = \\
    &=\frac{1}{|k|}
    \bigg| k|x|g\bigg(\frac{ax}{|ax|}\bigg)
    - k|x|g\bigg(\frac{x}{|x|}\bigg)
    \bigg|\\
    &= 0
\end{align*}
In the case $x = 0$ we see that $\lambda(k) = 0$ and hence
\begin{align*}
    &\frac{1}{|k|}
    \bigg||(a + k)x|g\bigg(\frac{(a + k)x}{|(a + k)x|}\bigg)
    - |ax|g\bigg(\frac{ax}{|ax|}\bigg)
    - k|x|g\bigg(\frac{x}{|x|}\bigg)
    \bigg| = \\
    &=\frac{1}{|k|} \bigg| 0 - 0 - 0 \bigg|\\
    &= 0
\end{align*}
Therefore $h$ is differentiable.
\item [(b)] Let us compute $Df(0,0)$ as follows
\begin{align*}
    &\lim_{(h,k) \to (0,0)} \frac{|f(0 + (h,k)) - f(0) - Df(0,0)(h,k)|}{|(h,k)|}\\
    &= \lim_{(h,k) \to (0,0)}
    \frac{1}{|(h,k)|}\bigg|
    |(h,k)|g\bigg(\frac{(h,k)}{|(h,k)|}\bigg)  - Df(0,0)(h,k)\bigg|
\end{align*}
Let $h = 0$ then
\begin{align*}
    &\frac{1}{|(0,k)|}\bigg|
    |(0,k)|g\bigg(\frac{(0,k)}{|(0,k)|}\bigg)  - Df(0,0)(0,k)\bigg|\\
    &=\frac{1}{|k|}\bigg|
    |k|g\bigg(0,\frac{k}{|k|}\bigg) - D_2f(0,0)k\bigg|\\
    &\leq \frac{1}{|k|}\bigg|
    |k|g\bigg(0,\frac{k}{|k|}\bigg)\bigg|
    + \frac{1}{|k|}\bigg|D_2f(0,0)k\bigg|\\
    &= \bigg|g\bigg(0,\frac{k}{|k|}\bigg)\bigg|
    + \bigg|D_2f(0,0)\bigg|
\end{align*}
We see that $g(0, k/|k|) = g(0, 1) = 0$ if $k > 0$, if $k < 0$ then
$g(0, k/|k|) = g(0, -1) = g(-(0, 1)) = -g(0,1) = 0$, then we get that
\begin{align*}
    0 \leq \frac{1}{|(0,k)|}\bigg|
    |(0,k)|g\bigg(\frac{(0,k)}{|(0,k)|}\bigg)  - Df(0,0)(0,k)\bigg|
    \leq \bigg|D_2f(0,0)\bigg|
\end{align*}
In the same way, if we let $k = 0$ instead
\begin{align*}
    &\frac{1}{|(h,0)|}\bigg|
    |(h,0)|g\bigg(\frac{(h,0)}{|(h,0)|}\bigg)  - Df(0,0)(h,0)\bigg|\\
    &=\frac{1}{|h|}\bigg|
    |h|g\bigg(\frac{h}{|h|},0\bigg) - D_1f(0,0)h\bigg|\\
    &\leq \frac{1}{|h|}\bigg|
    |h|g\bigg(\frac{h}{|h|},0\bigg)\bigg|
    + \frac{1}{|h|}\bigg|D_1f(0,0)h\bigg|\\
    &= \bigg|g\bigg(\frac{h}{|h|}, 0\bigg)\bigg|
    + \bigg|D_1f(0,0)\bigg|
\end{align*}
Then since $g(0,1) = 0$ we get that
\begin{align*}
    0 \leq \frac{1}{|(h,0)|}\bigg|
    |(h,0)|g\bigg(\frac{(h,0)}{|(h,0)|}\bigg)  - Df(0,0)(h,0)\bigg|
    \leq \bigg|D_1f(0,0)\bigg|
\end{align*}
Therefore must be that $Df(0,0) = 0$ for the limit 
\begin{align*}
    \lim_{(h,k) \to (0,0)} \frac{|f(0 + (h,k)) - f(0) - Df(0,0)(h,k)|}{|(h,k)|}
\end{align*}
to converge to 0.
\\
Then for any path taken by $h$ and $k$ the limit becomes
\begin{align*}
    &\lim_{(h,k) \to (0,0)} \frac{|f(0 + (h,k)) - f(0) - Df(0,0)(h,k)|}{|(h,k)|}\\
    &=\lim_{(h,k) \to (0,0)}
    \frac{1}{|(h,k)|}\bigg|
    |(h,k)|g\bigg(\frac{(h,k)}{|(h,k)|}\bigg)  - 0\bigg|\\
    &=\lim_{(h,k) \to (0,0)}
    \bigg|g\bigg(\frac{(h,k)}{|(h,k)|}\bigg)\bigg|
\end{align*}
But $g$ is only defined at the unit circle so this limit can only be 0 if
$g = 0$ and hence $f$ is not continuous at $(0,0)$.
\end{itemize}
\end{proof}

\begin{proof}{\textbf{2-5.}}
Let $f:\R^2 \to \R$ be defined by
\begin{align*}
    f(x,y) = \begin{cases}
        \frac{x|y|}{\sqrt{x^2 + y^2}} & (x,y) \neq 0\\
        0 & (x,y) = 0
    \end{cases}
\end{align*}
We can write that
\begin{align*}
    \frac{x|y|}{\sqrt{x^2 + y^2}}
    = \sqrt{x^2 + y^2}\cdot \frac{x|y|}{x^2 + y^2}
    = |(x,y)| \cdot \frac{x|y|}{|(x,y)|^2}
\end{align*}
Then defining $g$ as $g(x,y) = x|y|$ we see that
\begin{align*}
    g\bigg(\frac{(x,y)}{|(x,y)|}\bigg)
    = \frac{x}{|(x,y)|}\bigg|\frac{y}{|(x,y)|}\bigg|
    = \frac{x|y|}{|(x,y)|^2}
\end{align*}
Also, we see that $g(0,1) = g(1,0) = 0$ and that
\begin{align*}
    g(-x,-y) = -x|-y| = -xy = -x|y| = -g(x,y)   
\end{align*}
Hence we can write $f(x,y)$ as
\begin{align*}
    f(x,y) = \begin{cases}
        |(x,y)|\cdot g\bigg(\frac{(x,y)}{|(x,y)|}\bigg) & (x,y) \neq 0\\
        0 & (x,y) = 0
    \end{cases}
\end{align*}
Where $g$ is a continuous real-valued function with the same properties
described in Problem 2-4.
\\
Therefore we can conclude that $f$ is not differentiable at $(0,0)$.
\end{proof}

\begin{proof}{\textbf{2-6.}}
Let $f: \R^2 \to \R$ be defined by $f(x,y) = \sqrt{|xy|}$, we may also define
$f$ as
\begin{align*}
    f(x,y) = \begin{cases}
        \frac{\sqrt{x^2 + y^2}\sqrt{|xy|}}{\sqrt{x^2 + y^2}} & (x,y) \neq 0\\
        0 & (x,y) = 0
    \end{cases}
\end{align*}
We see that
\begin{align*}
    \frac{\sqrt{x^2 + y^2}\sqrt{|xy|}}{\sqrt{x^2 + y^2}}
    = |(x,y)| \frac{\sqrt{|xy|}}{|(x,y)|}
\end{align*}
We can define a function $g$ as $g(x,y) = \sqrt{|xy|}$ so
\begin{align*}
    g\bigg(\frac{(x,y)}{|(x,y)|}\bigg)
    = \sqrt{\bigg|\frac{x}{|(x,y)|}\frac{y}{|(x,y)|}\bigg|}
    = \sqrt{\frac{|xy|}{|(x,y)|^2}}
    = \frac{\sqrt{|xy|}}{|(x,y)|}
\end{align*}
Also, we see that $g(0,1) = g(1,0) = 0$ and that
\begin{align*}
    g(-x,-y) = \pm\sqrt{|(-x)(-y)|} = \pm\sqrt{|xy|} = \pm g(x,y)
\end{align*}
Hence we can write $f(x,y)$ as
\begin{align*}
    f(x,y) = \begin{cases}
        |(x,y)|\cdot g\bigg(\frac{(x,y)}{|(x,y)|}\bigg) & (x,y) \neq 0\\
        0 & (x,y) = 0
    \end{cases}
\end{align*}
Where $g$ is a continuous real-valued function with the same properties
described in Problem 2-4.
\\
Therefore we can conclude that $f$ is not differentiable at $(0,0)$.
\end{proof}

\begin{proof}{\textbf{2-7.}}
Let $f:\R^n \to \R$ be a function such that $|f(x)| \leq |x|^2$ we want to show
that $f$ is differentiable at 0.
\\
Let us consider the following limit
\begin{align*}
    \lim_{h \to 0} \frac{|f(0+h) - f(0) - Df(0)h|}{|h|}
\end{align*}
We see that
\begin{align*}
    0 \leq \frac{|f(0+h) - f(0) - Df(0)h|}{|h|}
    &\leq \frac{|f(h)| + |f(0)| + |Df(0)h|}{|h|}\\
    &\leq \frac{|h|^2 + 0 + |Df(0)||h|}{|h|}\\
    &\leq |h| + |Df(0)|
\end{align*}
We see that $|h| + |Df(0)| \to 0$ as $h \to 0$ if $Df(0) = 0$.
\\
Therefore $f$ is differnetiable at 0.
\end{proof}

\begin{proof}{\textbf{2-12}}\\
Let $f:\R^n \times \R^m \to \R^p$ be a bilinear function.
\begin{itemize}
\item[(a)] We want to prove that $\lim_{(h,k)\to 0}|f(h,k)|/|(h,k)| = 0$, we
note that
\begin{align*}
    f(h,k) = \sum_{i= 1}^n\sum_{j=1}^m h^ik^jf(e_i, e_j)
\end{align*}
Where $h^i$ and $k^j$ are the components of $h$ and $k$ respectively. Then
we see that
\begin{align*}
    \frac{|f(h,k)|}{|(h,k)|}
    &\leq \frac{1}{|(h,k)|}\sum_{i= 1}^n\sum_{j=1}^m |h^ik^jf(e_i, e_j)|
    \leq \frac{1}{|(h,k)|}\sum_{i= 1}^n\sum_{j=1}^m |h^i||k^j||f(e_i, e_j)|
\end{align*}
Let us take $M = \max_{i,j}{|f(e_i, e_j)|}$ then we get that
\begin{align*}
    \frac{|f(h,k)|}{|(h,k)|}
    &\leq \frac{1}{|(h,k)|}\sum_{i= 1}^n\sum_{j=1}^m |h^i||k^j||f(e_i, e_j)|
    \leq \frac{M}{|(h,k)|}\sum_{i= 1}^n\sum_{j=1}^m |h^i||k^j|
\end{align*}
Now, given that $|h^i||k^j| \leq |h||k|$ we have that
\begin{align*}
    M\frac{1}{|(h,k)|}\sum_{i= 1}^n\sum_{j=1}^m |h^i||k^j|
    &\leq M\frac{nm|h||k|}{\sqrt{|h|^2 + |k|^2}}\\
    &\leq M \frac{nm|h||k|}{\max\{|h|, |k|\}}\\
    &\leq M nm \min\{|h|,|k|\}
\end{align*}
Therefore following the chain of inequalities we get that
\begin{align*}
    0 \leq \lim_{(h,k)\to 0} \frac{|f(h,k)|}{|(h,k)|}
    \leq \lim_{(h,k) \to 0}M nm \min\{|h|,|k|\} = 0
\end{align*}
Hence
\begin{align*}
    \lim_{(h,k)\to 0} \frac{|f(h,k)|}{|(h,k)|} = 0
\end{align*}
\item[(b)] We want to prove that $Df(a,b)(x,y) = f(a,y) + f(x,b)$, then
see that
\begin{align*}
    &\frac{|f(a + h, b + k) - f(a,b) - Df(a,b)(h,k)|}{|(h,k)|} =\\
    &\quad= \frac{|f(a,b) + f(a, k) + f(h, b) + f(h,k) - f(a,b) - Df(a,b)(h,k)|}{|(h,k)|}\\
    &\quad= \frac{|f(a, k) + f(h, b) + f(h,k) - Df(a,b)(h,k)|}{|(h,k)|}\\
    &\quad= \frac{|f(a, k) + f(h, b) + f(h,k) - (f(a,k) + f(h,b))|}{|(h,k)|}\\
    &\quad= \frac{|f(h,k)|}{|(h,k)|}
\end{align*}
But we saw on part (a) that
\begin{align*}
    \lim_{(h,k)\to 0} \frac{|f(h,k)|}{|(h,k)|} = 0
\end{align*}
Therefore $Df(a,b)(x,y) = f(a,y) + f(x,b)$ is the derivative.
\item[(c)] From Theorem 2-3 if $p(x,y) = xy$ then we know that
\begin{align*}
    Dp(a,b)(x,y) = bx + ay
\end{align*}
But from part (b) we know that
\begin{align*}
    Df(a,b)(x,y) = f(x,b) + f(a,y)
\end{align*}
So taking $f = p$ we get that
\begin{align*}
    Dp(a,b)(x,y) = p(x,b) + p(a,y) = xb + ay
\end{align*}
Therefore the equation we have for $Dp(a,b)(x,y)$ from Theorem 2-3 is a special
case of the equation $Df(a,b)(x,y) = f(x,b) + f(a,y)$.
\end{itemize}
\end{proof}

\begin{proof}{\textbf{2-13}}\\
Let $IP:\R^n \times \R^n \to \R$ be defined as $IP(x,y)= \inprd{x,y}$
\begin{itemize}
\item [(a)] We see that if $x, x_1, x_2, y, y_1, y_2 \in \R^n$ and $a \in \R$
we have that
\begin{align*}
    \inprd{ax,y} &= a\inprd{x,y} = \inprd{x,ay}\\
    \inprd{x_1 + x_2,y} &= \inprd{x_1, y} + \inprd{x_2, y}\\
    \inprd{x,y_1 + y_2} &= \inprd{x, y_1} + \inprd{x, y_2}
\end{align*}
Then $IP$ is a bilinear function so from Problem 2-12 we have that
\begin{align*}
    D(IP(x,y))(a,b) = IP(a, y) + IP(x, b) = \inprd{a,y} + \inprd{x,b}
\end{align*}
To compute the Jacobian matrix $(IP(x,y))'(a,b)$ we assume that $IP(x,y)$ is
defined as $IP(x,y) = x_1y_1 + x_2y_2 + ... + x_ny_n$ where $x_i$ and $y_i$ are
the components of $x$ and $y$. Then 
\begin{align*}
    (IP(x,y))' = \begin{pmatrix}
        y_1, y_2, ..., y_n, x_1, x_2, ..., x_n
    \end{pmatrix}
\end{align*}
Hence
\begin{align*}
    (IP(x,y))'(a,b) &= \begin{pmatrix}
        y_1, y_2, ..., y_n, x_1, x_2, ..., x_n
    \end{pmatrix}
    \begin{pmatrix}
        a_1\\ ...\\ a_n\\ b_1\\ ...\\ b_n
    \end{pmatrix}\\
    &= a_1y_1 + ... + a_ny_n + b_1x_1 + ... + b_nx_n\\
    &= IP(a, y) + IP(x, b)
\end{align*}

\item [(b)] Let $f,g:\R \to \R^n$ be differentiable functions and let
$h:\R\to \R$ be defined by $h(t) = \inprd{f(t), g(t)}$.
\\
We can write $h$ as $h(t) = IP(f(t), g(t))$, so using the chain rule we get
that
\begin{align*}
    D(h(a))(v) &= D(IP(f(a),g(a)))(Df(a)(v), Dg(a)(v))\\
    &= IP(Df(a)(v),g(a)) + IP(f(a),Dg(a)(v))
\end{align*}
Since $f,g$ are differentiable then $Df(a)(v) = f'(a)(v)$ and
$Dg(a)(v) = g'(a)(v)$. Taking $v = 1$ we get that
\begin{align*}
    h'(a) &= \inprd{f'(a)^T, g(a)} + \inprd{f(a), g'(a)^T}
\end{align*}
Where we put $f'(a)^T$ and $g'(a)^T$ because we are considering
$IP: \R^n \times \R^n \to \R$ to be defined as $\inprd{x,y} = xy$.

\cleardoublepage
\item [(c)] Let $f:\R \to \R^n$ be differentiable and let $|f(t)| = 1$ for all
$t$.\\
Let us consider the function $h(t) = \inprd{f(t), f(t)}$, then we know that
\begin{align*}
    h'(a) = \inprd{f'(a)^T, f(a)} + \inprd{f(a), f'(a)^T}
\end{align*}
But also we know that
\begin{align*}
    h(t) = \inprd{f(t), f(t)} = |f(t)|^2 = 1
\end{align*}
So we get that $h'(a) = 0$ for any $a$, and hence
\begin{align*}
    0 &= \inprd{f'(a)^T, f(a)} + \inprd{f(a), f'(a)^T}\\

\end{align*}

\begin{align*}

    = \lim_{h\to 0} \frac{e^{-h^{-2}}}{h}
    = \lim_{h\to 0} \frac{1/h}{e^{h^{-2}}}
\end{align*}

\begin{align*}
    \lim_{h\to 0} \frac{1/h}{e^{h^{-2}}}
    = \lim_{h\to 0} \frac{-1/h^2}{-2e^{h^{-2}}/h^3}
    = \lim_{h\to 0} \frac{h}{2e^{h^{-2}}}
    = 0
\end{align*}

\begin{align*}
    f''(0) = \lim_{h\to 0} \frac{f'(0 + h) - f'(0)}{h}
    = \lim_{h\to 0} \frac{2e^{-h^{-2}}/h^3}{h}
    = \lim_{h\to 0} \frac{2/h^4}{e^{h^{-2}}}
\end{align*}

\begin{align*}
    \lim_{h\to 0} \frac{2/h^4}{e^{h^{-2}}}
    &= \lim_{h\to 0} \frac{-8/h^5}{-2e^{h^{-2}}/h^3}\\
    &= \lim_{h\to 0} \frac{4/h^2}{e^{h^{-2}}}\\
    &= \lim_{h\to 0} \frac{-8/h^3}{-2e^{h^{-2}}/h^3}\\
    &= \lim_{h\to 0} \frac{4}{e^{h^{-2}}}\\
    &= 0
\end{align*}
We see that every derivative of $e^{-x^{-2}}$ when $x\neq 0$ is of the form
\begin{align*}
    f^{(i)}(x) = \frac{P(x)}{Q(x)}e^{-x^{-2}}
\end{align*}

\end{align*}

that the growth of $e^{h^{-2}}$ is always faster than any polynomial, so the 
limit will always tend to $1/\infty = 0$.
\end{proof}

    &= 0
\end{align*}
Where we used the result we got from problem 2-25. Also, from problem 2-25 we
get that
\begin{align*}
    f^{(i)}(-1) = 0    
\end{align*}
For all $i$.\\
On the other hand, if $x = 1$ we get that
\begin{align*}
    f'(1) &= \lim_{h\to 0}\frac{f(1+h) - f(1)}{h}\\
    &= \lim_{h\to 0}\frac{e^{-h^{-2}} e^{-(h+2)^{-2}}}{h}\\
    &= \lim_{h\to 0}e^{-(h+2)^{-2}}\lim_{h\to 0}\frac{e^{-h^{-2}}}{h}\\
    &= e^{-x^{-4}}\lim_{h\to 0}\frac{e^{-h^{-2}}}{h}\\
    &= 0
\end{align*}
So we see that also for all $i$ we get that $f^{(i)}(1) = 0$.\\
Therefore the function $f$ is in $C^\infty$.
\\
If $x = 0$ we see that
\begin{align*}
    f(0) = e^{-(-1)^{-2}} e^{-(1)^{-2}}
    = \frac{1}{e^{\frac{1}{(-1)^2}}}\frac{1}{e^{\frac{1}{1^2}}}
    = \frac{1}{e^2} > 0
\end{align*}
If $x \in (-1, 1) \setminus \{0\}$ the function is given by
\begin{align*}
    f(x) = \frac{1}{e^{\frac{1}{(x-1)^2}}}\frac{1}{e^{\frac{1}{(x+1)^2}}}
\end{align*}
But $(x-1)^2$ and $(x + 1)^2$ are positive always so the function is always
positive here.
\\
Therefore the function $f$ is positive in the interval $(-1,1)$

\item [(b)] Let us consider the function
\begin{align*}
    g(x) = \frac{\int_0^x f}{\int_0^\epsilon f}
\end{align*}
Where $f$ is a function $C^\infty$ which is positive on $(0,\epsilon)$ and 0 
elsewhere.\\
If $x \leq 0$ we see that $\int_0^x f = 0$ and $\int_0^\epsilon f > 0$ then
$g(x) = 0$.\\
Now, suppose $x \geq \epsilon$ then $\int_0^x f = \int_0^\epsilon f$ since 
$f$ is positive on $(0,\epsilon)$ and 0 elsewere, hence
\begin{align*}
    g(x) = \frac{\int_0^x f}{\int_0^\epsilon f}
    = \frac{\int_0^\epsilon f}{\int_0^\epsilon f} = 1
\end{align*}
We need to prove now that $g$ is in $C^\infty$.\\
We note that
\begin{align*}
    g'(x) &= \frac{(\int_0^x f)'}{\int_0^\epsilon f}
    = \frac{f'}{\int_0^\epsilon f}
\end{align*}
Where we used the Fundamental Theorem of Calculus. Since $f \in C^\infty$ then
the expression for $g'$ is also infinitely differentiate, so $g \in C^\infty$.

\cleardoublepage
\item [(c)] Let $f$ be defined as 
\begin{align*}
    f(x) = \begin{cases}
        e^{-(x-1)^{-2}} e^{-(x+1)^{-2}} & x \in (-1,1)\\
        0 & x \not\in (-1, 1)
    \end{cases}
\end{align*}
Also, we define $g:\R^n \to \R$ as 
\begin{align*}
    g(x) = f([x^1 - a^1]/\epsilon) \cdot ... \cdot f([x^n - a^n]/\epsilon)
\end{align*}
Where $a \in \R^n$.
We know that $f$ is in $C^\infty$ because of part (a) then the
multiplication of $f$ n-times is in $C^\infty$ so $g$ is in $C^\infty$.
\\
Let $x \in (a^1 - \epsilon, a^1 + \epsilon) \times ... \times
(a^n - \epsilon, a^n + \epsilon)$ then we see that
\begin{align*}
    -1 \leq \frac{x^i - a^i}{\epsilon} \leq 1
\end{align*}
for $i \in \{1, ..., n\}$. We know that in the interval $(-1,1)$, $f$
is positive then so is $g$. \\
We know that elsewhere $f$ is 0 so $g$ is also 0 elsewhere.

\item [(d)] Let $A\subset \R^n$ be open and $C\subset A$ be compact.
Let us take for every point $c \in C$ an open box
$$(c^1 - \epsilon, c^1 + \epsilon) \times ... \times (c^n - \epsilon, c^n + \epsilon)$$
where $\epsilon$ is sufficiently small so the open boxes do not intersect with
the closed set $B$ where $f$ is supposed to be $0$.
\\
This is an open cover of $C$ and since $C$ is compact must be a finite subcover
that covers $C$ as well.
\\
Then we can define a set of functions $g_i: \cup_{i = 1}^m O_i \to \R$ over
the finite subcover $\cup_{i = 1}^m O_i$ such that $g_i$ is defined as
\begin{align*}
    g_i(x) = \begin{cases}
        f([x^1 - o_i^1]/\epsilon) \cdot ... \cdot f([x^n - o_i^n]/\epsilon)
        & x \in O_i\\
        0 & x \not\in O_i\\
    \end{cases}
\end{align*}
Where $o_i$ is the center of the open box $O_i$ and the function $f$ is defined
as in part (c).
\\
If we add all the functions we get another function $g: \cup_{i = 1}^m O_i \to \R$
defined as
\begin{align*}
    g(x) = \sum_{i=1}^m g_i(x)
\end{align*}
We note that $g$ is defined in an open set $A =\cup_{i = 1}^m O_i$, also $g$ is
in $C^\infty$ because each $g_i$ is in $C^\infty$ and $g(x) > 0$ for $x \in C$
but $g = 0$ at $B$, the closed set.

\cleardoublepage
\item [(e)] Since the function $g$ defined in part (d) has the property
$g(x) > 0$ for every $x \in C$ then we can always choose an $\epsilon > 0$
such that $g(x) \geq \epsilon$.
\\
So if we take the composed function $f = h \circ g$ where $h$ is defined as in
part (b) i.e. $h :\R \to [0,1]$ such that $h(x) = 0$ for $x \leq 0$ and
$h(x) = 1$ for $x\geq \epsilon$ then we see that $f = 1$ for every $x \in C$.
\end{itemize}
\end{proof}
